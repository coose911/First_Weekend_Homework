ls = list
ls -l = long list
ls -a = list incluiding hidden files
ls -la = long list including hidden files
~ - tilda = go back to home directory
cd .. = go back 1 space
cd (name the directory/file) = open directory or file
clear = clear screen
mkdir = make directory
touch = create file
pwd = print working directory/where am i 
rm = remove file
rmdir = remove empty directory
code (file name) = open file in IDE(VSC)
